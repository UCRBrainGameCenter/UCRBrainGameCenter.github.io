604  public int Read(float[] data, int offset, int count)
605  {
606  //When a read comes in, advance the system to the next whole tick, leaving
607  //buffered samples
608 
609  int samplesRemaining = count;
610 
611  int bodySamplesRead = ReadBuffered(data, offset, count);
612  samplesRemaining -= bodySamplesRead;
613  offset += bodySamplesRead;
614 
615  while (samplesRemaining > 0)
616  {
617  //Execute any Events at the current tick
618  while (NextEvent != null && NextEvent.time == currentTick)
619  {
620  NextEvent.ExecuteEvent(this);
621  eventIndex++;
622  }
623 
624  if (NextEvent == null)
625  {
626  break;
627  }
628 
629  int tickAdvance = Math.Min(
630  (int)Math.Ceiling(samplesRemaining / samplesPerTick),
631  NextEvent.time - currentTick);
632  double newSample = currentSample + tickAdvance * samplesPerTick;
633  int samplesInTick = (int)(newSample - currentSample);
634  int samplesToRead = Math.Min(samplesInTick, samplesRemaining);
635 
636  //NoteStream always returns requested samples
637  noteStream.Read(data, offset, samplesToRead);
638 
639  currentSample = newSample;
640  offset += samplesToRead;
641  samplesRemaining -= samplesToRead;
642  currentTick += tickAdvance;
643 
644  //Advance stream to the end of the tick if we still had tick samples left
645  if (samplesInTick > samplesToRead)
646  {
647  bufferCount = samplesInTick - samplesToRead;
648 
649  noteStream.Read(sampleBuffer, 0, bufferCount);
650  bufferIndex = 0;
651  }
652  }
653 
654  return count - samplesRemaining;
655  }
