802  {

804  sampleSquaredSum[1] += samples[2 * i + 1] * samples[2 * i + 1];
805  }
806

808  sampleSquaredSum[1] = Math.Sqrt(sampleSquaredSum[1] / sampleCount);

811  double rmsR = sampleSquaredSum[1];

819  }
