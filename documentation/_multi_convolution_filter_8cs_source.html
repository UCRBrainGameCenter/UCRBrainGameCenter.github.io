196  public override int Read(float[] data, int offset, int count)
197  {
198  if (!initialized)
199  {
200  Initialize();
201  }
202 
203  int samplesWritten = ReadBody(data, offset, count);
204 
205  while (samplesWritten < count)
206  {
207  int read = stream.Read(inputBuffer, 0, inputBuffer.Length);
208 
209  if (read <= 0)
210  {
211  //Done, No samples left to work with
212  break;
213  }
214 
215  //Slide output samples over to accumulate on the remainder
216  Array.Copy(
217  sourceArray: outputAccumulation,
218  sourceIndex: bufferCount,
219  destinationArray: outputAccumulation,
220  destinationIndex: 0,
221  length: outputAccumulation.Length - bufferCount);
222 
223  Array.Clear(
224  array: outputAccumulation,
225  index: outputAccumulation.Length - bufferCount,
226  length: bufferCount);
227 
228  bufferIndex = 0;
229  bufferCount = Channels * samplesPerOverlap;
230 
231 
232  if (read < inputBuffer.Length)
233  {
234  //Copy all the remaining samples
235  //We are guaranteed to have enough room because the output buffer's
236  //length is Channels * (inputBuffer.Length + fftLength)
237  bufferCount = Channels * (read + filterLength - 1);
238  //Set rest of inputBuffer to zero
239  Array.Clear(inputBuffer, read, inputBuffer.Length - read);
240  }
241 
242  for (int i = 0; i < samplesPerOverlap; i++)
243  {
244  fftBuffer[i] = inputBuffer[i];
245  }
246 
247  Array.Clear(fftBuffer, samplesPerOverlap, fftLength - samplesPerOverlap);
248 
249  //FFT
250  Fourier.Forward(fftBuffer);
251 
252  for (int channel = 0; channel < Channels; channel++)
253  {
254  for (int i = 0; i < fftLength; i++)
255  {
256  ifftBuffer[i] = fftBuffer[i] * filterFD[channel][i];
257  }
258

261

266  }
267  }
